param(
  [string]$AccessFile = ".\App_Data\TrackerData.mdb",
  [string]$OutJson = ".\Docs\Generated\AccessSchema.json",
  [string]$OutProfile = ".\Docs\Reports\AccessProfile.tsv"

Add-Type -AssemblyName System.Data
$cs = "Provider=Microsoft.ACE.OLEDB.12.0;Data Source=$AccessFile;Persist Security Info=False;"
$cn = New-Object System.Data.OleDb.OleDbConnection($cs); $cn.Open()

$tables = $cn.GetOleDbSchemaTable([System.Data.OleDb.OleDbSchemaGuid]::Tables, @($null,$null,$null,"TABLE"))
$result = @()
foreach ($row in $tables.Rows) {
  $tName = $row.TABLE_NAME
  $cols = $cn.GetOleDbSchemaTable([System.Data.OleDb.OleDbSchemaGuid]::Columns, @($null,$null,$tName,$null))
  $cObjs = @()
  foreach ($c in $cols.Rows) {
    $cObjs += [pscustomobject]@{
      ColumnName = $c.COLUMN_NAME
      Ordinal    = $c.ORDINAL_POSITION
      DataType   = $c.DATA_TYPE
      Width      = $c.CHARACTER_MAXIMUM_LENGTH
      Nullable   = ($c.IS_NULLABLE -eq "YES")
    }
  }
  $cmd = $cn.CreateCommand(); $cmd.CommandText = "SELECT COUNT(*) FROM [$tName]"
  $count = $cmd.ExecuteScalar()
  $result += [pscustomobject]@{ Table=$tName; RowCount=$count; Columns=$cObjs }
}
$cn.Close()
$result | ConvertTo-Json -Depth 7 | Out-File $OutJson -Encoding UTF8
"Table`tColumn`tRowCount" | Out-File $OutProfile -Encoding UTF8
$result | ForEach-Object { $_.Columns | ForEach-Object { "$($_.Table)`t$($_.ColumnName)`t$($_.RowCount)" | Add-Content $OutProfile } }
Write-Host "Schema exported to $OutJson"

PHASE 4 – DATA MIGRATION (REPEATABLE SCRIPT)
===============================================================================
Goal: Move ALL data Access -> SQL preserving identity & row counts; record validation.

Script: Database\DataMigration.ps1 (bulk copy KeepIdentity).

Outline (create file):
-----------------------------------------------------------------------------
param(
  [string]$AccessFile = ".\App_Data\TrackerData.mdb",
  [string]$SqlServer = ".\SQLEXPRESS",
  [string]$Database = "TrackerData",
  [switch]$Clear
)

Add-Type -AssemblyName System.Data
$accessCs = "Provider=Microsoft.ACE.OLEDB.12.0;Data Source=$AccessFile;Persist Security Info=False;"
$sqlCs    = "Server=$SqlServer;Database=$Database;Integrated Security=SSPI;MultipleActiveResultSets=True;"

$tables = @(
  # Order parents first (adjust to real list)
  "ServiceTypesTbl",
  "ItemTypeTbl",
  "CityTbl",
  "PackagingTbl",
  "PriceLevelsTbl",
  "PaymentTermsTbl",
  "InvoiceTypeTbl",
  "RepairStatusesTbl"
  # add remaining tables...
)

$log = ".\Docs\Reports\MigrationRun_$(Get-Date -Format yyyyMMdd_HHmmss).log"
"Migration start $(Get-Date -Format u)" | Out-File $log

$cnA = New-Object System.Data.OleDb.OleDbConnection($accessCs); $cnA.Open()

foreach ($t in $tables) {
  Write-Host "Migrating $t"
  if ($Clear) { sqlcmd -S $SqlServer -d $Database -Q "DELETE FROM [$t]" | Out-Null }
  $cmd = $cnA.CreateCommand(); $cmd.CommandText = "SELECT * FROM [$t]"
  $da = New-Object System.Data.OleDb.OleDbDataAdapter($cmd)
  $dt = New-Object System.Data.DataTable
  [void]$da.Fill($dt)

  if ($dt.Rows.Count -gt 0) {
    $bulk = New-Object Data.SqlClient.SqlBulkCopy($sqlCs,[Data.SqlClient.SqlBulkCopyOptions]::KeepIdentity)
    $bulk.DestinationTableName = $t
    foreach ($c in $dt.Columns) { $null = $bulk.ColumnMappings.Add($c.ColumnName,$c.ColumnName) }
    $bulk.WriteToServer($dt)
  }

  $sqlCount = sqlcmd -S $SqlServer -d $Database -h -1 -W -Q "SET NOCOUNT ON; SELECT COUNT(*) FROM [$t];"
  "$t`tin:$($dt.Rows.Count)`tout:$sqlCount" | Add-Content $log
}

$cnA.Close()
"Migration end $(Get-Date -Format u)" | Add-Content $log
Write-Host "Finished. Log: $log"
